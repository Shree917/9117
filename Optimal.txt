#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n, frames;
    cout << "Enter number of pages: ";
    cin >> n;
    vector<int> pages(n);
    cout << "Enter page reference string: ";
    for (int i = 0; i < n; i++) cin >> pages[i];
    cout << "Enter number of frames: ";
    cin >> frames;

    vector<int> memory;
    int hits = 0, faults = 0;

    for (int i = 0; i < n; i++) {
        int page = pages[i];
        auto it = find(memory.begin(), memory.end(), page);

        if (it != memory.end()) {
            hits++;
        } else {
            if (memory.size() < frames)
                memory.push_back(page);
            else {
                int farthest = -1, index = -1;
                for (int j = 0; j < memory.size(); j++) {
                    int k;
                    for (k = i + 1; k < n; k++)
                        if (memory[j] == pages[k]) break;
                    if (k > farthest) {
                        farthest = k;
                        index = j;
                    }
                }
                memory[index] = page;
            }
            faults++;
        }
    }

    cout << "\nTotal Page Faults: " << faults;
    cout << "\nTotal Page Hits: " << hits;
    cout << "\nPage Fault Ratio: " << (float)faults / n * 100 << "%";
    cout << "\nPage Hit Ratio: " << (float)hits / n * 100 << "%\n";
}
